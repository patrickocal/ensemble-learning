 \begin{itemize}
\item<+-|alert@+>{\color<.>{blue}\emph{
By risk neutrality, if $x\equiv(a,s)$,  $ x'\equiv (a,s')$, $y\equiv (b,t)$ and $y'\equiv(b,t')$, then the first moments satisfy:
}} 
\uncover<+-|alert@+>{\color<.>{blue}
\[
\text{ $\phi_{xy}(\lambda)\sim \phi_{x'y'}(\lambda)\sim a+\lambda (b-a)$
}\]
}\vskip-20pt
\item<+-|alert@+>{\color<.>{blue}\emph{
Second moments, if $a=b$ and $t-s=t'-s'$, then
}}
\uncover<+-|alert@+>{\color<.>{blue}\emph{
\[
\text{ $\left(\phi_{xy}(\lambda)\right)^2\sim \left(\phi_{x'y'}(\lambda)\right)^2\sim \left(\lambda-\lambda^2\right)(t-s)$
}\]
}}
\uncover<+-|alert@+>{\color<.>{blue}\emph{
and standardised higher moments are all zero,
}}\uncover<+-|alert@+>{\color<.>{blue}\emph{
  then Val's beliefs are characterised by Wiener measure.\normalsize
}}
 \end{itemize}
\uncover<+-|alert@+>{\color<.>{blue}
To account for risk averse preferences, need a continuous function $u:\R\rightarrow \R$ such that $u\circ \phi_{xy}$ satisfies the moment conditions.
}
\end{frame}
